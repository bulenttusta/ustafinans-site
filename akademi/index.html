document.addEventListener('DOMContentLoaded', () => { const tabs = document.querySelectorAll('.tab'); const panels = { student: document.getElementById('student-panel'), company: document.getElementById('company-panel') }; function switchTab(targetTab) { tabs.forEach(tab => { const isActive = tab.dataset.tab === targetTab; tab.classList.toggle('active', isActive); tab.setAttribute('aria-selected', isActive); }); Object.keys(panels).forEach(key => { const panel = panels[key]; const shouldShow = key === targetTab; panel.style.display = shouldShow ? 'block' : 'none'; // Opsiyonel: geçiş animasyonu için class ekleyebilirsin // panel.classList.toggle('fade-in', shouldShow); }); } tabs.forEach(tab => { tab.addEventListener('click', () => { switchTab(tab.dataset.tab); }); // Klavye erişilebilirliği için Enter/Space desteği tab.addEventListener('keydown', e => { if (e.key === 'Enter' || e.key === ' ') { e.preventDefault(); switchTab(tab.dataset.tab); } }); }); });
